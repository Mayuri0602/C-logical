#include<stdio.h>
int main(){
    char m;
    printf("enter the value of m");
    scanf(" %c ",&m);
    if (m=='a'){
        int e=1;
        while(e<=5){
            printf(" %d ",e);
        }
        e++;;
    }
    else if (m=='b'){
        int f=5;
        while(f>=1){
            printf(" %d ",f);
        }
        f--;
    }
    else if (m=='c'){
        int g=45;
        while(g<=89){
            if (g%2==0){
                printf(" %d ",g);
            }
            g++;
        }
    }
    else if (m=='d'){
        int h=78;
        while(h>=34){
            if (h%2==1){
                printf(" %d ",h);
            }
            h--;
            }
    }
    return 0;
}
